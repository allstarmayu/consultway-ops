# 05 — Database Schema

> SQLite (Cloudflare D1) schema, authored in Drizzle ORM.
> This document describes the logical model. The source of truth is
> `src/db/schema.ts` — do not edit this file without updating schema.ts, and
> don't edit schema.ts without regenerating migrations.

## Overview

Tables grouped by domain:

- **Identity:** `users`
- **Companies:** `companies`, `company_contacts`, `company_sectors`
- **Documents:** `documents`, `reminders_sent`
- **Tenders:** `tenders`, `tender_applications`, `tender_eligibility_rules`
- **Projects (Phase 3):** `projects`, `milestones`, `project_activity`
- **Finance (Phase 3):** `transactions`
- **Platform:** `notifications`, `audit_log`

## D1 / SQLite Conventions

Because D1 is SQLite, we use these conventions project-wide:

| Rule | Reason |
|---|---|
| Primary keys are `TEXT` UUIDs (`crypto.randomUUID()`) | Distributed-safe; no auto-increment collisions across replicas |
| Timestamps are `INTEGER` unix seconds OR `TEXT` ISO strings | We use TEXT ISO for readability in Drizzle; convert in app |
| "Enums" are `TEXT` with a `CHECK` constraint | SQLite has no native enum |
| JSON fields are `TEXT` with `{ mode: 'json' }` in Drizzle | Drizzle auto-parses; SQLite has no JSONB |
| Monetary amounts are `INTEGER` **paise** | Never store money as REAL/FLOAT |
| Booleans are `INTEGER` (0/1) | SQLite has no boolean type |
| Foreign keys ON, cascade where safe | `PRAGMA foreign_keys = ON` at migration time |
| Every table has `created_at` and `updated_at` | Audit + debugging |
| `updated_at` maintained via a trigger OR via app (we do it via app + Payload hooks) | Keep SQL portable |

## Schema — Identity

### `users`

Every person who logs in has a `User`. Companies are linked via a join:
a company can have multiple user accounts (POC + finance contact, etc).

| Column | Type | Constraints | Notes |
|---|---|---|---|
| `id` | TEXT | PRIMARY KEY | UUID v4 |
| `email` | TEXT | UNIQUE NOT NULL | Lowercased on insert |
| `email_verified_at` | TEXT | NULLABLE | ISO timestamp |
| `password_hash` | TEXT | NOT NULL | bcrypt, cost 10 |
| `full_name` | TEXT | NOT NULL |  |
| `role` | TEXT | NOT NULL CHECK (role IN ('admin','staff','company-user')) |  |
| `company_id` | TEXT | NULLABLE, FK → companies.id ON DELETE SET NULL | Only for company-users |
| `phone` | TEXT | NULLABLE |  |
| `avatar_url` | TEXT | NULLABLE | R2 public URL |
| `last_login_at` | TEXT | NULLABLE | ISO timestamp |
| `status` | TEXT | NOT NULL DEFAULT 'active' CHECK (status IN ('active','disabled')) |  |
| `created_at` | TEXT | NOT NULL DEFAULT (current ISO) |  |
| `updated_at` | TEXT | NOT NULL DEFAULT (current ISO) |  |

Indexes:
- `idx_users_email` on `email`
- `idx_users_company_id` on `company_id`
- `idx_users_role` on `role`

## Drizzle Schema File (skeleton)

```ts
// src/db/schema.ts
/**
 * Consultway Ops — database schema (Cloudflare D1 / SQLite).
 *
 * Conventions:
 *   - All IDs are UUIDs stored as TEXT
 *   - Timestamps are ISO-8601 strings stored as TEXT
 *   - "Enums" are TEXT with a matching CHECK constraint
 *   - Money is stored in INTEGER paise (never REAL)
 */
import {
  sqliteTable,
  text,
  integer,
  index,
  uniqueIndex,
  primaryKey,
  check,
} from 'drizzle-orm/sqlite-core';
import { sql, relations } from 'drizzle-orm';

// ─────────────────────── helpers ───────────────────────
const id = () => text('id').primaryKey().$defaultFn(() => crypto.randomUUID());
const createdAt = () => text('created_at').notNull().$defaultFn(() => new Date().toISOString());
const updatedAt = () => text('updated_at').notNull().$defaultFn(() => new Date().toISOString());

// ─────────────────────── users ───────────────────────
export const users = sqliteTable('users', {
  id: id(),
  email: text('email').notNull(),
  emailVerifiedAt: text('email_verified_at'),
  passwordHash: text('password_hash').notNull(),
  fullName: text('full_name').notNull(),
  role: text('role', { enum: ['admin', 'staff', 'company-user'] }).notNull(),
  companyId: text('company_id').references(() => companies.id, { onDelete: 'set null' }),
  phone: text('phone'),
  avatarUrl: text('avatar_url'),
  lastLoginAt: text('last_login_at'),
  status: text('status', { enum: ['active', 'disabled'] }).notNull().default('active'),
  createdAt: createdAt(),
  updatedAt: updatedAt(),
}, (t) => [
  uniqueIndex('idx_users_email').on(t.email),
  index('idx_users_company_id').on(t.companyId),
  index('idx_users_role').on(t.role),
]);

// ─────────────────────── companies ───────────────────────
export const companies = sqliteTable('companies', {
  id: id(),
  legalName: text('legal_name').notNull(),
  displayName: text('display_name'),
  cin: text('cin').notNull(),
  gstin: text('gstin'),
  pan: text('pan'),
  status: text('status', { enum: ['pending', 'active', 'suspended', 'rejected'] })
    .notNull().default('pending'),
  rejectionReason: text('rejection_reason'),
  addressLine1: text('address_line1').notNull(),
  addressLine2: text('address_line2'),
  city: text('city').notNull(),
  state: text('state').notNull(),
  pincode: text('pincode').notNull(),
  website: text('website'),
  yearsInBusiness: integer('years_in_business'),
  annualTurnoverPaise: integer('annual_turnover_paise'),
  verifiedAt: text('verified_at'),
  verifiedBy: text('verified_by').references(() => users.id),
  registeredAt: text('registered_at').notNull(),
  createdAt: createdAt(),
  updatedAt: updatedAt(),
}, (t) => [
  uniqueIndex('idx_companies_cin').on(t.cin),
  index('idx_companies_status').on(t.status),
  index('idx_companies_legal_name').on(t.legalName),
]);

// … (remainder of tables follow the same pattern)

// ─────────────────────── relations ───────────────────────
export const usersRelations = relations(users, ({ one, many }) => ({
  company: one(companies, { fields: [users.companyId], references: [companies.id] }),
  notifications: many(notifications),
}));

export const companiesRelations = relations(companies, ({ many }) => ({
  users: many(users),
  documents: many(documents),
  applications: many(tenderApplications),
  projects: many(projects),
}));
```
